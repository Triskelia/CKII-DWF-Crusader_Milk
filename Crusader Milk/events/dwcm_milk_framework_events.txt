#.010 Initialize hucow related variables in case it wasn't already (.ie trait added with console) 
character_event = {
	id = DWCMMilk.010
	hide_window = yes
	only_women = yes
	min_age = 16

	is_triggered_only = yes

	trigger = {
		trait = hucow
		NOT = { has_character_flag = hucow_variables_initialized }
	}

	immediate = {
		dwcm_set_base_milk_capacity = yes
		dwcm_set_base_milk_production_rate = yes
		dwcm_set_base_milk_production_multiplier = yes
		dwcm_set_total_milk_production_rate = yes
		dwcm_set_total_milk_capacity= yes
		set_variable = { which = dwcm_breast_fullness value = 0 }
	}

	option = { 
		set_character_flag = hucow_variables_initialized
  	}
}
