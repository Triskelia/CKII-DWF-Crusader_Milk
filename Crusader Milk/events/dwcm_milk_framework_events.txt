#.010 Monthly current_milk_quantity increase
character_event = {
	id = DWCMMilk.010
	hide_window = yes
	only_women = yes
	min_age = 16

	# is_triggered_only = yes
	mean_time_to_happen = { months = 1 }

	trigger = {
		trait = hucow
	}

	immediate = {
		if = {
			limit = { dwcm_is_non_initialized_hucow = yes }
			dwcm_become_hucow = yes # We initialize the hucow caracteristics if they aren't already (ie. if trait was added with console command)
		}
		dwcm_increase_current_milk_quantity = yes
	}
}
